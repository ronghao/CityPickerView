# 使用方法
1.在根 build.gradle中添加

    allprojects {
        repositories {
            ...
            maven { url 'https://jitpack.io' }
        }
    }

2.在项目build.gradle中添加 ，版本参考 ![](https://jitpack.io/v/ronghao/CityPickerView.svg)

    dependencies {
        compile 'com.github.ronghao:CityPickerView:1.0.0'
    }
3.
